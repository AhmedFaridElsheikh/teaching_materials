import pandas as pd
from sklearn.datasets import load_iris

iris_raw = load_iris()
df = pd.DataFrame(iris_raw.data, columns=['sepal_length', 'sepal_width', 'petal_length', 'petal_width'])
df['species'] = iris_raw.target_names[iris_raw.target]
